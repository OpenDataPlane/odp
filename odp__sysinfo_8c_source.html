434 static void print_cipher_capa(odp_cipher_alg_t cipher)
435 {
436  int caps = odp_crypto_cipher_capability(cipher, NULL, 0);
437  int rc, i;
438 
439  if (caps <= 0)
440  return;
441 
442  odp_crypto_cipher_capability_t capa[caps];
443 
444  rc = odp_crypto_cipher_capability(cipher, capa, caps);
445  if (rc < 0)
446  return;
447 
448  printf(" %s:\n", cipher_alg_name(cipher));
449  for (i = 0; i < rc; i++)
450  printf(" key %d iv %d\n",
451  capa[i].key_len, capa[i].iv_len);
452 }

455 {

459  if (caps <= 0)

472  printf(" key %d", capa[i].key_len);
473  if (capa[i].iv_len != 0)
474  printf(" iv %d", capa[i].iv_len);
475  if (capa[i].aad_len.max != 0)
476  printf(" aad %d, %d, %d",
477  capa[i].aad_len.min, capa[i].aad_len.max,
478  capa[i].aad_len.inc);
479  printf("\n");

529 static int pktio_capability(appl_args_t *appl_args)
530 {
531  odp_pool_param_t pool_param;
532  odp_pool_t pool;
533  int ret = 0;
534 
535  odp_pool_param_init(&pool_param);
536 
537  pool_param.type = ODP_POOL_PACKET;
538  pool_param.pkt.num = 128;
539 
540  pool = odp_pool_create("pktio_pool", &pool_param);
541  if (pool == ODP_POOL_INVALID) {
542  ODPH_ERR("Creating packet pool failed\n");
543  return -1;
544  }
545
