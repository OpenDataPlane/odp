442 static void print_cipher_capa(odp_cipher_alg_t cipher)
443 {
444  int caps = odp_crypto_cipher_capability(cipher, NULL, 0);
445  int rc, i;
446 
447  if (caps <= 0)
448  return;
449 
450  odp_crypto_cipher_capability_t capa[caps];
451 
452  rc = odp_crypto_cipher_capability(cipher, capa, caps);
453  if (rc < 0)
454  return;
455 
456  printf(" %s:\n", cipher_alg_name(cipher));
457  for (i = 0; i < rc; i++)
458  printf(" key %d iv %d\n",
459  capa[i].key_len, capa[i].iv_len);
460 }

462 static void print_auth_capa(odp_auth_alg_t auth)
463 {
464  int caps = odp_crypto_auth_capability(auth, NULL, 0);
465  int rc, i;
466 
467  if (caps <= 0)
468  return;
469 
470  odp_crypto_auth_capability_t capa[caps];
471 
472  rc = odp_crypto_auth_capability(auth, capa, caps);
473  if (rc < 0)
474  return;
475 
476  printf(" %s:\n", auth_alg_name(auth));
477  for (i = 0; i < rc; i++) {
478  printf(" digest %d", capa[i].digest_len);

480  printf(" key %d", capa[i].key_len);
481  if (capa[i].iv_len != 0)
482  printf(" iv %d", capa[i].iv_len);
483  if (capa[i].aad_len.max != 0)
484  printf(" aad %d, %d, %d",
485  capa[i].aad_len.min, capa[i].aad_len.max,
486  capa[i].aad_len.inc);
