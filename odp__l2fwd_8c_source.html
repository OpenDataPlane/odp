332 /*
333  * Drop packets which input parsing marked as containing errors.
334  *
335  * Frees packets with error and modifies pkt_tbl[] to only contain packets with
336  * no detected errors.
337  *
338  * pkt_tbl Array of packets
339  * num Number of packets in pkt_tbl[]
340  *
341  * Returns number of packets dropped
342  */
343 static inline int drop_err_pkts(odp_packet_t pkt_tbl[], unsigned num)
344 {
345  odp_packet_t pkt;
346  unsigned dropped = 0;
347  unsigned i, j;
348 
349  for (i = 0, j = 0; i < num; ++i) {
350  pkt = pkt_tbl[i];
351 
352  if (odp_unlikely(odp_packet_has_error(pkt))) {
353  odp_packet_free(pkt); /* Drop */
354  dropped++;
355  } else if (odp_unlikely(i != j++)) {
356  pkt_tbl[j - 1] = pkt;
357  }
358  }
359 
360  return dropped;
361 }

587 static inline void handle_tx_poll_compl(tx_compl_t *tx_c, odp_packet_t pkts[], int num, int tx_idx,
588  stats_t *stats)
589 {
590  uint32_t num_act = tx_c->num_act, poll_head = tx_c->poll_head, free_head = tx_c->free_head;
591  const uint32_t max = tx_c->max, init = tx_c->init, max_act = tx_c->max_act;
592  odp_pktio_t pktio = gbl_args->pktios[tx_idx].pktio;
593  int next_req = tx_c->next_req;
594  odp_packet_t pkt;
595  const int interval = tx_c->interval;
596 
597  while (num_act > 0) {
598  if (odp_packet_tx_compl_done(pktio, poll_head) < 1)
599  break;
600 
601  --num_act;
602 
603  if (++poll_head > max)
604  poll_head = init;
605  }
606 
607  while (next_req <= num) {
608  pkt = pkts[next_req - 1];
609 
610  if (num_act == max_act) {
611  stats->s.tx_c_misses++;
612  /* Missed one, try requesting for the first packet of next burst. */
613  next_req = num + 1;
614  break;
615  }
616 
617  tx_c->opt.compl_id = free_head;
618 
619  if (odp_packet_tx_compl_request(pkt, &tx_c->opt) < 0) {
620  stats->s.tx_c_fails++;
621  /* Missed one, try requesting for the first packet of next burst. */
622  next_req = num + 1;
623  break;
624  }
625 
626  if (++free_head > max)
627  free_head = init;
628 
629  ++num_act;
630  next_req += interval;
631  }
632 
633  tx_c->free_head = free_head;
634  tx_c->poll_head = poll_head;
635  tx_c->num_act = num_act;
636  tx_c->next_req = next_req - num;
637 }

662 static inline void send_packets(odp_packet_t *pkt_tbl,
663  int pkts,
664  int use_event_queue,
665  int tx_idx,
666  odp_queue_t tx_queue,
667  odp_pktout_queue_t pktout_queue,
668  state_t *state,
669  stats_t *stats)
670 {
671  int sent;
672  unsigned int tx_drops;
673  int i;
674  odp_packet_t pkt;
675 
676  if (odp_unlikely(state != NULL))
677  handle_tx_state(state, pkt_tbl, pkts, tx_idx, stats);
678 
679  if (odp_unlikely(use_event_queue))
680  sent = event_queue_send(tx_queue, pkt_tbl, pkts);
681  else
682  sent = odp_pktout_send(pktout_queue, pkt_tbl, pkts);
683 
684  sent = odp_unlikely(sent < 0) ? 0 : sent;
685  tx_drops = pkts - sent;
686 
687  if (odp_unlikely(tx_drops)) {
688  stats->s.tx_drops += tx_drops;

806  odp_packet_vector_free(pkt_vec);
807  continue;
808  }

829  * Such packets can block sending of later packets in other threads
830  * that then would never enter the thread barrier and we would
831  * end up in a dead-lock.
832  */
833  odp_schedule_pause();

835  odp_event_t ev;
836 
837  ev = odp_schedule(NULL, ODP_SCHED_NO_WAIT);
838  if (ev == ODP_EVENT_INVALID)
839  break;
840 
841  odp_event_free(ev);
842  }
843 
844  /* Make sure that latest stat writes are visible to other threads */
845  odp_mb_full();
846 
847  /* Wait until pktio devices are stopped */
848  odp_barrier_wait(&gbl_args->term_barrier);
849 
850  /* Free remaining events in queues */
851  odp_schedule_resume();
852  while (1) {
853  odp_event_t ev;
854 
855  ev = odp_schedule(NULL,
856  odp_schedule_wait_time(ODP_TIME_SEC_IN_NS));
857 
858  if (ev == ODP_EVENT_INVALID)
859  break;
860 
861  odp_event_free(ev);
862  }
863 
864  return 0;
865 }

1735  *
1736  * less workers (N) than interfaces (M)
1737  * - assign each worker to process every Nth interface

1742  * - interfaces are processed by inequal number of workers, when N is not
1743  * divisible by M
1744  * - tries to configure a queue per worker per interface
1745  * - shares queues, if interface capability does not allows a queue per worker

1754  num_workers = gbl_args->appl.num_workers;
1755 
1756  if (gbl_args->appl.sched_mode) {
1757  /* all threads receive and send on all pktios */
1758  for (i = 0; i < if_count; i++) {
1759  gbl_args->pktios[i].num_rx_thr = num_workers;
1760  gbl_args->pktios[i].num_tx_thr = num_workers;
1761  }
1762 
1763  for (thr = 0; thr < num_workers; thr++) {
1764  thr_args = &gbl_args->thread_args[thr];
1765  thr_args->num_pktio = if_count;
1766 
1767  /* In sched mode, pktios are not cross connected with
1768  * local pktio indexes */
