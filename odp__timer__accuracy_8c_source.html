413 static int single_shot_params(test_global_t *test_global, odp_timer_pool_param_t *timer_param,
414  odp_timer_capability_t *timer_capa)
415 {
416  uint64_t res_ns, res_hz;
417  uint64_t max_res_ns, max_res_hz;
418  uint64_t period_ns = test_global->opt.period_ns;
419  uint64_t num_tmo = test_global->opt.num + test_global->opt.num_warmup;
420  uint64_t offset_ns = test_global->opt.offset_ns;
421  enum mode_e mode = test_global->opt.mode;
422 
423  max_res_ns = timer_capa->max_res.res_ns;
424  max_res_hz = timer_capa->max_res.res_hz;
425 
426  /* Default resolution */
427  if (test_global->opt.res_ns == 0 && test_global->opt.res_hz == 0) {
428  res_ns = test_global->opt.period_ns / 10;
429  res_hz = 0;
430  } else if (test_global->opt.res_ns) {
431  res_ns = test_global->opt.res_ns;
432  res_hz = 0;
433  } else {
434  res_ns = 0;
435  res_hz = test_global->opt.res_hz;
436  }
437 
438  if (res_ns && res_ns < max_res_ns) {
439  ODPH_ERR("Resolution %" PRIu64 " nsec too high.\n"
440  "Highest resolution %" PRIu64
441  " nsec. Default resolution is period / 10.\n\n",
442  res_ns, max_res_ns);
443  return -1;
444  }
445 
446  if (res_hz && res_hz > max_res_hz) {
447  ODPH_ERR("Resolution %" PRIu64 " hz too high.\n"
448  "Highest resolution %" PRIu64
449  " hz. Default resolution is period / 10.\n\n",
450  res_hz, max_res_hz);
451  return -1;
452  }
453 
454  if (res_ns)
455  timer_param->res_ns = res_ns;
456  else
457  timer_param->res_hz = res_hz;
458 
459  if (mode == MODE_ONESHOT) {
460  timer_param->min_tmo = offset_ns / 2;
461  timer_param->max_tmo = offset_ns + ((num_tmo + 1) * period_ns);
462  } else {
463  if (mode == MODE_RESTART_ABS)
464  timer_param->min_tmo = period_ns / 10;
465  else
466  timer_param->min_tmo = period_ns;
467  timer_param->max_tmo = offset_ns + (2 * period_ns);
468  }
469 
470  if (test_global->opt.max_tmo_ns) {
471  if (test_global->opt.max_tmo_ns < timer_param->max_tmo) {
472  ODPH_ERR("Max tmo is too small. Must be at least %" PRIu64 " nsec.\n",
473  timer_param->max_tmo);
474  return -1;
475  }
476 
477  timer_param->max_tmo = test_global->opt.max_tmo_ns;
478  }
479 
480  printf(" period: %" PRIu64 " nsec\n", period_ns);
481  printf(" max res nsec: %" PRIu64 "\n", max_res_ns);
482  printf(" max res hertz: %" PRIu64 "\n", max_res_hz);
483 
484  test_global->period_dbl = period_ns;
485 
486  return 0;
487 }

489 static int periodic_params(test_global_t *test_global, odp_timer_pool_param_t *timer_param,
490  odp_timer_capability_t *timer_capa)
491 {
492  int ret;
493  uint64_t res_ns;
494  odp_timer_periodic_capability_t capa;
495  double freq_dbl, min_freq, max_freq;
496  double opt_freq = odp_fract_u64_to_dbl(&test_global->opt.freq);
497  odp_fract_u64_t freq = test_global->opt.freq;
498  uint64_t res_hz = test_global->opt.res_hz;
499  uint64_t max_multiplier = test_global->opt.max_multiplier;
500  uint64_t multiplier = test_global->opt.multiplier;
501 
502  if (res_hz) {
503  res_ns = ODP_TIME_SEC_IN_NS / res_hz;
504  } else {
505  res_ns = test_global->opt.res_ns;
506 
507  /* Default resolution */
508  if (res_ns == 0)
509  res_ns = ODP_TIME_SEC_IN_NS / (10 * multiplier * opt_freq);
510  }
511 
512  if (res_ns == 0) {
513  ODPH_ERR("Resolution too high\n");
514  return -1;
515  }
516 
517  /* Resolution from capa */
518  if (test_global->opt.res_ns < 0)
519  res_ns = 0;
520 
521  min_freq = odp_fract_u64_to_dbl(&timer_capa->periodic.min_base_freq_hz);
522  max_freq = odp_fract_u64_to_dbl(&timer_capa->periodic.max_base_freq_hz);
523 
524  capa.base_freq_hz = freq;
525  capa.max_multiplier = max_multiplier;
526  capa.res_ns = res_ns;
527 
528  ret = odp_timer_periodic_capability(test_global->opt.clk_src, &capa);
529 
530  if (ret < 0) {
531  ODPH_ERR("Requested periodic timer capabilities are not supported.\n"
532  "Capabilities: min base freq %g Hz, max base freq %g Hz, "
533  "max res %" PRIu64 " Hz\n",
534  min_freq, max_freq, timer_capa->max_res.res_hz);
535  return -1;
536  }
537 
538  if (ret == 0) {
539  printf("Requested base frequency is not met. Using %.2f Hz instead of %.2f Hz.\n",
540  odp_fract_u64_to_dbl(&capa.base_freq_hz), opt_freq);
541 
542  freq = capa.base_freq_hz;
543  }
544 
545  if (res_ns == 0)
546  res_ns = capa.res_ns;
547 
548  freq_dbl = odp_fract_u64_to_dbl(&freq);
549  test_global->base_freq = freq;
550  test_global->period_dbl = ODP_TIME_SEC_IN_NS / (multiplier * freq_dbl);
551 
552  /* Min/max tmo are ignored, leave those to default values */
553  timer_param->timer_type = ODP_TIMER_TYPE_PERIODIC;
554  timer_param->periodic.base_freq_hz = freq;
555  timer_param->periodic.max_multiplier = max_multiplier;
556 
557  if (res_hz)
558  timer_param->res_hz = res_hz;
559  else
560  timer_param->res_ns = res_ns;
561 
562  printf(" min freq capa: %.2f hz\n", min_freq);
563  printf(" max freq capa: %.2f hz\n", max_freq);
564  printf(" freq option: %.2f hz\n", opt_freq);
565  printf(" freq: %.2f hz\n", freq_dbl);
566  printf(" freq integer: %" PRIu64 "\n", freq.integer);
567  printf(" freq numer: %" PRIu64 "\n", freq.numer);
568  printf(" freq denom: %" PRIu64 "\n", freq.denom);
569  printf(" max_multiplier: %" PRIu64 "\n", max_multiplier);
570  printf(" multiplier: %" PRIu64 "\n", multiplier);
571  printf(" timer freq: %.2f hz\n", multiplier * freq_dbl);
572  printf(" timer period: %.2f nsec\n", test_global->period_dbl);
573  printf(" resolution capa: %" PRIu64 " nsec\n", capa.res_ns);
574 
575  return 0;
576 }

796 static int start_timers(test_global_t *test_global)
797 {
798  odp_timer_pool_t timer_pool;

800  uint64_t period_ns, start_ns, nsec, offset_ns;
801  odp_time_t time;
802  uint64_t i, j, idx, num_tmo, num_warmup, burst, burst_gap;
803  enum mode_e mode;
804 
805  mode = test_global->opt.mode;
806  num_warmup = test_global->opt.num_warmup;

809  burst_gap = test_global->opt.burst_gap;
810  period_ns = test_global->opt.period_ns;
811  offset_ns = test_global->opt.offset_ns;
812  timer_pool = test_global->timer_pool;
813  idx = 0;
814 
815  /* Record test start time and tick. Memory barriers forbid compiler and out-of-order
816  * CPU to move samples apart. */
817  odp_mb_full();
818  start_tick = odp_timer_current_tick(timer_pool);
819  time = odp_time_global();
820  odp_mb_full();
821 
822  start_ns = odp_time_to_ns(time);
823  test_global->start_tick = start_tick;
824  test_global->start_ns = start_ns;
825  test_global->period_tick = odp_timer_ns_to_tick(timer_pool, period_ns);
826 
827  /* When mode is not one-shot, set only one burst of timers initially */
828  if (mode != MODE_ONESHOT)
829  num_tmo = 1;
830 
831  for (i = 0; i < num_tmo; i++) {
832  odp_timer_retval_t retval;
833 
834  for (j = 0; j < burst; j++) {
835  timer_ctx_t *ctx = &test_global->timer_ctx[idx];
836  odp_timer_start_t start_param;
837 
838  if (mode == MODE_PERIODIC) {
839  odp_timer_periodic_start_t periodic_start;
840 
841  nsec = offset_ns + (j * burst_gap);
842 
843  /* By default, timer starts one period after current time. Round
844  * floating point to closest integer number. */
845  ctx->nsec = start_ns + test_global->period_dbl + 0.5;
846  if (nsec)
847  ctx->nsec = start_ns + nsec;
848 
849  ctx->first_period = start_tick +
850  odp_timer_ns_to_tick(timer_pool,
851  test_global->period_dbl + 0.5);
852  periodic_start.freq_multiplier = test_global->opt.multiplier;
853  periodic_start.first_tick = 0;
854  if (nsec)
855  periodic_start.first_tick =
856  start_tick + odp_timer_ns_to_tick(timer_pool, nsec);
857  periodic_start.tmo_ev = ctx->event;
858  retval = odp_timer_periodic_start(ctx->timer, &periodic_start);
859  } else if (mode == MODE_CONCURRENCY) {
860  ctx->nsec = start_ns + test_global->opt.period_ns;
861  start_param.tick_type = ODP_TIMER_TICK_REL;
862  start_param.tick = test_global->period_tick;
863  start_param.tmo_ev = ctx->event;
864  ctx->starts++;
865  retval = odp_timer_start(ctx->timer, &start_param);
866  } else {
867  nsec = offset_ns + (i * period_ns) + (j * burst_gap);
868  ctx->nsec = start_ns + nsec;
869  start_param.tick_type = ODP_TIMER_TICK_ABS;
870  start_param.tick =
871  start_tick + odp_timer_ns_to_tick(timer_pool, nsec);
872  start_param.tmo_ev = ctx->event;
873  retval = odp_timer_start(ctx->timer, &start_param);
874  }
875 
876  if (retval != ODP_TIMER_SUCCESS) {
877  ODPH_ERR("Timer[%" PRIu64 "] set failed: %i\n", idx, retval);
878  return -1;
879  }
880 
881  idx++;
882  }
883  }
884 
885  printf("\nStarting timers took %" PRIu64 " nsec\n", odp_time_global_ns() - start_ns);
886 
887  return 0;
888 }

1249  ODPH_ERR("odp_timer_start(): %d\n", ret);
1250  goto error;
1251  }
1252  }
1253  }
1254

1256  odp_ticketlock_unlock(&ctx->lock);

1262  odp_ticketlock_unlock(&ctx->lock);
