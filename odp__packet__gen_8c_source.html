384 static int parse_vlan(const char *str, test_global_t *global)
385 {
386  struct vlan_hdr *vlan;
387  const char *start = str;
388  char *end;
389  int num_vlan = 0;
390  intptr_t str_len = strlen(str);
391 
392  while (num_vlan < MAX_VLANS) {
393  vlan = &global->test_options.vlan[num_vlan];
394 
395  /* TPID in hexadecimal */
396  end = NULL;
397  vlan->tpid = strtoul(start, &end, 16);
398  if (end < start)
399  break;
400 
401  /* Skip ':' */
402  start = end + 1;
403  if (start - str >= str_len)
404  break;
405 
406  /* TCI in hexadecimal */
407  end = NULL;
408  vlan->tci = strtoul(start, &end, 16);
409  if (end < start)
410  break;
411 
412  num_vlan++;
413 
414  /* Skip ',' or stop at the string end */
415  start = end + 1;
416  if (start - str >= str_len)
417  break;
418  }
419 
420  return num_vlan;
421 }

423 static inline uint64_t bswap(uint64_t in, uint32_t len)
424 {
425  uint8_t byte;
426  uint64_t result = 0;
427 
428  if (ODP_BIG_ENDIAN || len == sizeof(uint8_t))
429  return in;
430 
431  for (uint32_t i = 0; i < len; i++) {
432  byte = (in >> (8 * i)) & 0xff;
433  result |= ((uint64_t)byte << (8 * (len - 1 - i)));
434  }
435 
436  return result;
437 }

439 static odp_bool_t parse_custom_fields(const char *optarg, test_options_t *opts)
440 {
441  char *tmp_str = strdup(optarg), *tmp;
442  uint32_t num_fields = 0;
443  char name[MAX_HDR_NAME_LEN + 1];
444  uint32_t len;
445  uint64_t value;
446  int64_t diff;
447  hdr_field_t *hdr;
448  int ret;
449 
450  if (tmp_str == NULL)
451  return false;
452 
453  tmp = strtok(tmp_str, TOKEN_DELIMITER);
454 
455  if (tmp == NULL) {
456  free(tmp_str);
457  return false;
458  }
459 
460  opts->custom_l3.eth_type = strtoul(tmp, NULL, 16);
461  tmp = strtok(NULL, TOKEN_DELIMITER);
462 
463  while (tmp) {
464  if (num_fields == MAX_HDR_FIELDS) {
465  ODPH_ERR("Invalid custom header, too many fields: %u\n", num_fields + 1);
466  free(tmp_str);
467  return false;
468  }
469 
470  ret = sscanf(tmp, "%" S(MAX_HDR_NAME_LEN) "[^" FIELD_DELIMITER "]" FIELD_DELIMITER
471  "%u" FIELD_DELIMITER "%" PRIx64 FIELD_DELIMITER "%" PRIi64 "", name,
472  &len, &value, &diff);
473 
474  if (ret != 4) {
475  ODPH_ERR("Invalid custom header, bad field format\n");
476  free(tmp_str);
477  return false;
478  }
479 
480  if (len > MAX_HDR_VALUE_SZ) {
481  ODPH_ERR("Invalid custom header, field length too long: %u\n", len);
482  free(tmp_str);
483  return false;
484  }
485 
486  hdr = &opts->custom_l3.fields[num_fields];
487  odph_strcpy(hdr->name, name, MAX_HDR_NAME_LEN + 1);
488  hdr->value = value;
489  hdr->diff = diff;
490  hdr->len = len;
491  opts->custom_l3.tot_len += len;
492  ++num_fields;
493  tmp = strtok(NULL, TOKEN_DELIMITER);
494  }
495 
496  opts->num_custom_l3 = num_fields;
497  free(tmp_str);
498 
499  return true;
500 }

502 static odp_bool_t parse_lso_fields(const char *optarg, test_options_t *opts)
503 {
504  char *tmp_str = strdup(optarg), *tmp;
505  uint8_t num_custom = 0;
506  uint32_t proto, mod_op, offset, size;
507  int ret;
508  odp_bool_t ret_val = true;
509 
510  odp_lso_profile_param_init(&opts->lso.param);
511 
512  if (tmp_str == NULL) {
513  ODPH_ERR("Error: strdup() failed\n");
514  return false;
515  }
516 
517  /* Format: proto,payload_offset,max_payload_len[,mod_op:offset:size,mod_op:offset...] */
518 
519  tmp = strtok(tmp_str, TOKEN_DELIMITER);
520  if (tmp == NULL) {
521  ODPH_ERR("Error: Unable to parse LSO protocol\n");
522  ret_val = false;
523  goto exit;
524  }
525 
526  proto = strtoul(tmp, NULL, 0);
527 
528  switch (proto) {
529  case 0:
530  opts->lso.param.lso_proto = ODP_LSO_PROTO_IPV4;
531  break;
532  case 1:
533  opts->lso.param.lso_proto = ODP_LSO_PROTO_TCP_IPV4;
534  break;
535  case 2:
536  opts->lso.param.lso_proto = ODP_LSO_PROTO_CUSTOM;
537  break;
538  default:
539  ODPH_ERR("Error: Invalid LSO protocol: %u\n", proto);
540  ret_val = false;
541  goto exit;
542  }
543 
544  tmp = strtok(NULL, TOKEN_DELIMITER);
545  if (tmp == NULL) {
546  ODPH_ERR("Error: Unable to parse LSO payload offset\n");
547  ret_val = false;
548  goto exit;
549  }
550  opts->lso.payload_offset = strtoul(tmp, NULL, 0);
551 
552  tmp = strtok(NULL, TOKEN_DELIMITER);
553  if (tmp == NULL) {
554  ODPH_ERR("Error: Unable to parse LSO max payload length\n");
555  ret_val = false;
556  goto exit;
557  }
558  opts->lso.max_payload_len = strtoul(tmp, NULL, 0);
559 
560  tmp = strtok(NULL, TOKEN_DELIMITER);
561  while (opts->lso.param.lso_proto == ODP_LSO_PROTO_CUSTOM && tmp) {
562  if (num_custom == ODP_LSO_MAX_CUSTOM) {
563  ODPH_ERR("Error: Too many custom LSO operations (max %u)\n",
564  ODP_LSO_MAX_CUSTOM);
565  ret_val = false;
566  goto exit;
567  }
568 
569  ret = sscanf(tmp, "%" PRIu32 "" FIELD_DELIMITER
570  "%" PRIu32 "" FIELD_DELIMITER "%" PRIu32 "",
571  &mod_op, &offset, &size);
572 
573  if (ret != 3) {
574  ODPH_ERR("Error: Invalid custom LSO operation, bad field format\n");
575  ret_val = false;
576  goto exit;
577  }
578 
579  switch (mod_op) {
580  case 0:
581  opts->lso.param.custom.field[num_custom].mod_op = ODP_LSO_ADD_SEGMENT_NUM;
582  break;
583  case 1:
584  opts->lso.param.custom.field[num_custom].mod_op = ODP_LSO_ADD_PAYLOAD_LEN;
585  break;
586  case 2:
587  opts->lso.param.custom.field[num_custom].mod_op =
588  ODP_LSO_ADD_PAYLOAD_OFFSET;
589  break;
590  default:
591  ODPH_ERR("Error: Invalid custom LSO operation: %" PRIu32 "\n", mod_op);
592  ret_val = false;
593  goto exit;
594  }
595 
596  opts->lso.param.custom.field[num_custom].offset = offset;
597 
598  if (size != 1 && size != 2 && size != 4 && size != 8) {
599  ODPH_ERR("Error: Invalid custom field size: %" PRIu32 "\n", size);
600  ret_val = false;
601  goto exit;
602  }
603  opts->lso.param.custom.field[num_custom].size = size;
604 
605  num_custom++;
606  tmp = strtok(NULL, TOKEN_DELIMITER);
607  }
608 
609  if (opts->lso.param.lso_proto == ODP_LSO_PROTO_CUSTOM && num_custom == 0) {

1109 static int set_num_cpu(test_global_t *global)
1110 {
1111  int ret;
1112  test_options_t *test_options = &global->test_options;
1113  int num_cpu = test_options->num_cpu;
1114 
1115  ret = odp_cpumask_default_worker(&global->cpumask, num_cpu);
1116 
1117  if (ret != num_cpu) {
1118  int cpu;
1119 
1120  /* Normally we want to use only worker threads */
1121  if (ret > 1) {
1122  ODPH_ERR("Error: Too many workers. Maximum supported %i.\n", ret);
1123  return -1;
1124  }
1125 
1126  /* When number of workers is very limited (e.g. ODP project CI),
1127  * we try to use any CPUs available. */
1128  ret = odp_cpumask_all_available(&global->cpumask);
1129  if (ret < num_cpu) {
1130  ODPH_ERR("Error: Not enough CPUs. Maximum supported %i.\n", ret);
1131  return -1;
1132  }
1133 
1134  /* Remove extra CPUs from the mask */
1135  cpu = odp_cpumask_first(&global->cpumask);
1136  while (ret > num_cpu) {
1137  odp_cpumask_clr(&global->cpumask, cpu);
1138  cpu = odp_cpumask_first(&global->cpumask);
1139  ret--;
1140  }
1141  }
1142 
1143  odp_barrier_init(&global->barrier, num_cpu + 1);
1144 
1145  return 0;
1146 }

1148 static int check_lso_capa(char *name, odp_pktio_capability_t *pktio_capa,
1149  test_options_t *opt, odp_pool_t pool)
1150 {
1151  odp_packet_t pkt;
1152  uint32_t max_segments;
1153  uint32_t pkt_payload;
1154  const uint32_t max_pkt_len = opt->use_rand_pkt_len ?
1155  opt->rand_pkt_len_max : opt->pkt_len;
1156 
1157  if (pktio_capa->lso.max_profiles < opt->num_pktio ||
1158  pktio_capa->lso.max_profiles_per_pktio == 0) {
1159  ODPH_ERR("Error (%s): Not enough LSO profiles (max %" PRIu32 ")\n", name,
1160  ODPH_MIN(pktio_capa->lso.max_profiles,
1161  pktio_capa->lso.max_profiles_per_pktio));
1162  return -1;
1163  }
1164 
1165  if (opt->lso.param.lso_proto == ODP_LSO_PROTO_TCP_IPV4 && !pktio_capa->lso.proto.tcp_ipv4) {
1166  ODPH_ERR("Error (%s): ODP_LSO_PROTO_TCP_IPV4 not supported\n", name);
1167  return -1;
1168  }
1169  if (opt->lso.param.lso_proto == ODP_LSO_PROTO_IPV4 && !pktio_capa->lso.proto.ipv4) {
1170  ODPH_ERR("Error (%s): ODP_LSO_PROTO_IPV4 not supported\n", name);
1171  return -1;
1172  }
1173  if (opt->lso.param.lso_proto == ODP_LSO_PROTO_CUSTOM && !pktio_capa->lso.proto.custom) {
1174  ODPH_ERR("Error (%s): ODP_LSO_PROTO_CUSTOM not supported\n", name);
1175  return -1;
1176  }
1177 
1178  /* Check number of segments in a max length packet */
1179  pkt = odp_packet_alloc(pool, max_pkt_len);
1180  if (pkt == ODP_PACKET_INVALID) {
1181  ODPH_ERR("Error (%s): Allocating test packet failed\n", name);
1182  return -1;
1183  }
1184 
1185  max_segments = odp_packet_num_segs(pkt);
