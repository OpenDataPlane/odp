683 static int
684 create_session_from_config(odp_crypto_session_t *session,
685  crypto_alg_config_t *config,
686  crypto_args_t *cargs)
687 {
688  odp_crypto_session_param_t params;
689  odp_crypto_ses_create_err_t ses_create_rc;
690  odp_queue_t out_queue;
691 
692  memcpy(&params, &config->session, sizeof(odp_crypto_session_param_t));
693  params.op = ODP_CRYPTO_OP_ENCODE;
694  params.op_type = ODP_CRYPTO_OP_TYPE_BASIC;
695  params.auth_cipher_text = true;
696  params.output_pool = ODP_POOL_INVALID;
697 
698  if (cargs->schedule || cargs->poll) {
699  out_queue = odp_queue_lookup("crypto-out");
700  if (out_queue == ODP_QUEUE_INVALID) {
701  ODPH_ERR("crypto-out queue not found\n");
702  return -1;
703  }
704  params.compl_queue = out_queue;
705  params.op_mode = ODP_CRYPTO_ASYNC;
706  } else {
707  params.compl_queue = ODP_QUEUE_INVALID;
708  params.op_mode = ODP_CRYPTO_SYNC;
709  }
710  if (odp_crypto_session_create(&params, session,
711  &ses_create_rc)) {
712  switch (ses_create_rc) {
713  case ODP_CRYPTO_SES_ERR_ALG_COMBO:
714  printf(" requested algorithm combination not supported\n");
715  return 1;
716  case ODP_CRYPTO_SES_ERR_ALG_ORDER:
717  printf(" requested algorithm order not supported\n");
718  return 1;
719  case ODP_CRYPTO_SES_ERR_PARAMS:
720  printf(" requested session parameters not supported\n");
721  return 1;
722  default:
723  break;
724  }
725  ODPH_ERR("crypto session create failed.\n");
726  return -1;
727  }
728 
729  return 0;
730 }
