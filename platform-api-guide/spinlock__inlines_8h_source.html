37 _ODP_INLINE void odp_spinlock_lock(odp_spinlock_t *spinlock)
38 {
39  /* While the lock is already taken... */
40  while (__atomic_test_and_set(&spinlock->lock, __ATOMIC_ACQUIRE))
41  /* ...spin reading the flag (relaxed MM),
42  * the loop will exit when the lock becomes available
43  * and we will retry the TAS operation above */
44  while (__atomic_load_n(&spinlock->lock, __ATOMIC_RELAXED))
45  odp_cpu_pause();
46 }

48 _ODP_INLINE int odp_spinlock_trylock(odp_spinlock_t *spinlock)
49 {
50  return (__atomic_test_and_set(&spinlock->lock, __ATOMIC_ACQUIRE) == 0);
51 }

53 _ODP_INLINE void odp_spinlock_unlock(odp_spinlock_t *spinlock)
54 {
55  __atomic_clear(&spinlock->lock, __ATOMIC_RELEASE);
56 }
