36_ODP_INLINE void odp_ticketlock_init(odp_ticketlock_t *ticketlock)
37{
38 odp_atomic_init_u32(&ticketlock->next_ticket, 0);
39 odp_atomic_init_u32(&ticketlock->cur_ticket, 0);
40}
41
42_ODP_INLINE void odp_ticketlock_lock(odp_ticketlock_t *ticketlock)
43{
44 uint32_t ticket;
45
46 /* Take a ticket using an atomic increment of 'next_ticket'.
47 * This can be a relaxed operation but it cannot have the
48 * acquire semantics since we haven't acquired the lock yet */
49 ticket = odp_atomic_fetch_inc_u32(&ticketlock->next_ticket);
50
51 /* Spin waiting for our turn. Use load-acquire so that we acquire
52 * all stores from the previous lock owner */
53 _odp_wait_until_equal_acq_u32(&ticketlock->cur_ticket, ticket);
54}

56_ODP_INLINE int odp_ticketlock_trylock(odp_ticketlock_t *tklock)
57{
58 /* We read 'next_ticket' and 'cur_ticket' non-atomically which should
59 * not be a problem as they are not independent of each other.
60 * 'cur_ticket' is always <= to 'next_ticket' and if we see an
61 * older value of 'cur_ticket', this only means the lock will
62 * look busy and trylock will fail. */
63 uint32_t next = odp_atomic_load_u32(&tklock->next_ticket);
64 uint32_t cur = odp_atomic_load_u32(&tklock->cur_ticket);
65 /* First check that lock is available and possible to take without
66 * spinning. */
67 if (next == cur) {
68 /* Then try to take the lock by incrementing 'next_ticket'
69 * but only if it still has the original value which is
70 * equal to 'cur_ticket'.
71 * We don't have to include 'cur_ticket' in the comparison
72 * because it cannot be larger than 'next_ticket' (only
73 * smaller if the lock is busy).
74 * If CAS fails, it means some other thread intercepted and
75 * took a ticket which means the lock is not available
76 * anymore */
77 if (odp_atomic_cas_acq_u32(&tklock->next_ticket,
78 &next, next + 1))
79 return 1;
80 }
81 return 0;
82}

84_ODP_INLINE void odp_ticketlock_unlock(odp_ticketlock_t *ticketlock)
85{
86 /* Release the lock by incrementing 'cur_ticket'. As we are the
87 * lock owner and thus the only thread that is allowed to write
88 * 'cur_ticket', we don't need to do this with an (expensive)
89 * atomic RMW operation. Instead load-relaxed the current value
90 * and a store-release of the incremented value */
91 uint32_t cur = odp_atomic_load_u32(&ticketlock->cur_ticket);
92
93 odp_atomic_store_rel_u32(&ticketlock->cur_ticket, cur + 1);
94}
95
96_ODP_INLINE int odp_ticketlock_is_locked(odp_ticketlock_t *ticketlock)
97{
98 /* Compare 'cur_ticket' with 'next_ticket'. Ideally we should read
99 * both variables atomically but the information can become stale
100 * immediately anyway so the function can only be used reliably in
101 * a quiescent system where non-atomic loads should not pose a
102 * problem */
103 return odp_atomic_load_u32(&ticketlock->cur_ticket) !=
104 odp_atomic_load_u32(&ticketlock->next_ticket);
105}
