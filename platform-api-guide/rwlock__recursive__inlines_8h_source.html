48 _ODP_INLINE void odp_rwlock_recursive_read_lock(odp_rwlock_recursive_t *rlock)
49 {
50  int thr = odp_thread_id();
51 
52  if (rlock->rd_cnt[thr]) {
53  _ODP_ASSERT(rlock->rd_cnt[thr] < UINT8_MAX);
54  rlock->rd_cnt[thr]++;
55  return;
56  }
57 
58  odp_rwlock_read_lock(&rlock->lock);
59  rlock->rd_cnt[thr] = 1;
60 }

81 _ODP_INLINE void odp_rwlock_recursive_read_unlock(odp_rwlock_recursive_t *rlock)
82 {
83  int thr = odp_thread_id();
84 
85  _ODP_ASSERT(rlock->rd_cnt[thr]);
86  rlock->rd_cnt[thr]--;
87 
88  if (rlock->rd_cnt[thr] > 0)
89  return;

94 /* Only one writer can recurse the lock */
95 _ODP_INLINE void odp_rwlock_recursive_write_lock(odp_rwlock_recursive_t *rlock)
96 {
97  int thr = odp_thread_id();
98 
99  if (rlock->wr_owner == thr) {
100  _ODP_ASSERT(rlock->wr_cnt < UINT32_MAX);
101  rlock->wr_cnt++;
102  return;
103  }
104 
105  odp_rwlock_write_lock(&rlock->lock);
106  rlock->wr_owner = thr;
107  rlock->wr_cnt = 1;
108 }

130 _ODP_INLINE void odp_rwlock_recursive_write_unlock(odp_rwlock_recursive_t *rlock)
131 {
132  _ODP_ASSERT(rlock->wr_cnt);
133  rlock->wr_cnt--;
134 
135  if (rlock->wr_cnt > 0)
136  return;
137 
138  rlock->wr_owner = -1;
139  odp_rwlock_write_unlock(&rlock->lock);
