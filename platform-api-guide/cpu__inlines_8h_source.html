70 _ODP_INLINE uint64_t odp_cpu_cycles_diff(uint64_t c2, uint64_t c1)
71 {
72  if (odp_likely(c2 >= c1))
73  return c2 - c1;
74 
75  return c2 + (odp_cpu_cycles_max() - c1) + _odp_cpu_cycles_resolution();
76 }

80  /*

83  * rw 0..1 (0: read, 1: write)
