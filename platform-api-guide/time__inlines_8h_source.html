101 _ODP_INLINE int odp_time_cmp(odp_time_t t2, odp_time_t t1)
102 {
103  if (odp_likely(t2.u64 > t1.u64))
104  return 1;
105 
106  if (t2.u64 < t1.u64)
107  return -1;
108 
109  return 0;
