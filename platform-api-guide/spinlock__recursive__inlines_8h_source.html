44 _ODP_INLINE void odp_spinlock_recursive_lock(odp_spinlock_recursive_t *rlock)
45 {
46  int thr = odp_thread_id();
47 
48  if (rlock->owner == thr) {
49  _ODP_ASSERT(rlock->cnt < UINT32_MAX);
50  rlock->cnt++;
51  return;
52  }
53 
54  odp_spinlock_lock(&rlock->lock);
55  rlock->owner = thr;
56  rlock->cnt = 1;
57 }

59 _ODP_INLINE int odp_spinlock_recursive_trylock(odp_spinlock_recursive_t *rlock)
60 {
61  int thr = odp_thread_id();
62 
63  if (rlock->owner == thr) {
64  _ODP_ASSERT(rlock->cnt < UINT32_MAX);
65  rlock->cnt++;
66  return 1;
67  }
68 
69  if (odp_spinlock_trylock(&rlock->lock)) {
70  rlock->owner = thr;
71  rlock->cnt = 1;
72  return 1;
73  }
74 
75  return 0;
76 }

78 _ODP_INLINE void odp_spinlock_recursive_unlock(odp_spinlock_recursive_t *rlock)
79 {
80  _ODP_ASSERT(rlock->cnt);
81  rlock->cnt--;
82 
83  if (rlock->cnt > 0)
84  return;
85 
86  rlock->owner = -1;
87  odp_spinlock_unlock(&rlock->lock);
88 }

90 _ODP_INLINE int odp_spinlock_recursive_is_locked(odp_spinlock_recursive_t *rlock)
91 {
92  return odp_thread_id() == rlock->owner ? 1 : odp_spinlock_is_locked(&rlock->lock);
93 }
