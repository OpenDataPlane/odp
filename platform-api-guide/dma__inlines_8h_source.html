76 _ODP_INLINE void odp_dma_compl_free(odp_dma_compl_t dma_compl)
77 {
78  int8_t *ev_type;
79  odp_event_t ev;
80  odp_buffer_t buf = (odp_buffer_t)(uintptr_t)dma_compl;
81 
82  if (odp_unlikely(dma_compl == ODP_DMA_COMPL_INVALID)) {
83  _ODP_ERR("Bad DMA compl handle\n");
84  return;
85  }
86 
87  ev = odp_buffer_to_event(buf);
88  ev_type = _odp_event_hdr_ptr(ev, int8_t, event_type);
89  *ev_type = ODP_EVENT_BUFFER;
90 
91  odp_buffer_free(buf);
92 }
