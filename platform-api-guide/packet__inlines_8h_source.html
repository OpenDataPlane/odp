284 _ODP_INLINE void *odp_packet_l2_ptr(odp_packet_t pkt, uint32_t *len)
285 {
286  uint32_t offset = odp_packet_l2_offset(pkt);
287  uint32_t seg_len = odp_packet_seg_len(pkt);
288  uint8_t *data = (uint8_t *)odp_packet_data(pkt);
289 
290  if (odp_unlikely(offset >= seg_len)) {
291  void *pkt_hdr = (void *)pkt;
292 
293  return _odp_packet_map(pkt_hdr, offset, len, NULL);
294  }
295 
296  if (len)
297  *len = seg_len - offset;
298 
299  return data + offset;
300 }

366 
367  input_flags.all = _odp_pkt_get(pkt, uint64_t, input_flags);

369  if (input_flags.tcp)
370  return ODP_PROTO_L4_TYPE_TCP;

374  return ODP_PROTO_L4_TYPE_SCTP;

381  else if (input_flags.icmp && input_flags.ipv6)
382  return ODP_PROTO_L4_TYPE_ICMPV6;
383  else if (input_flags.no_next_hdr)
384  return ODP_PROTO_L4_TYPE_NO_NEXT;
385 
386  return ODP_PROTO_L4_TYPE_NONE;
