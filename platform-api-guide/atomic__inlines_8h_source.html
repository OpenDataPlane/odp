216 #define _ODP_ATOMIC_CAS_OP(ret_ptr, old_val, new_val) \
217 __extension__ ({ \
218  if (atom->v == (old_val)) { \
219  atom->v = (new_val); \
220  *(ret_ptr) = 1; \
221  } else { \
222  *(ret_ptr) = 0; \
223  } \
224 })
225 
233 #define _ODP_ATOMIC_OP(atom, expr) \
234 __extension__ ({ \
235  uint64_t _old_val; \
236  /* Loop while lock is already taken, stop when lock becomes clear */ \
237  while (__atomic_test_and_set(&(atom)->lock, __ATOMIC_ACQUIRE)) \
238  (void)0; \
239  _old_val = (atom)->v; \
240  (expr); /* Perform whatever update is desired */ \
241  __atomic_clear(&(atom)->lock, __ATOMIC_RELEASE); \
242  _old_val; /* Return old value */ \
243 })
